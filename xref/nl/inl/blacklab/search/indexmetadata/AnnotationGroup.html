1   package nl.inl.blacklab.search.indexmetadata;

3   import java.util.ArrayList;
4   import java.util.Iterator;
5   import java.util.List;
6   import java.util.stream.Collectors;
7   import java.util.stream.Stream;
8   
9   /** Logical grouping of annotations */
10  public class AnnotationGroup implements Iterable<Annotation> {
11      
12      private AnnotatedFieldsImpl annotatedFields;
13      
14      private String fieldName;
15      
16      private String groupName;
17      
18      private List<String> annotations;
19      
20      private boolean addRemainingAnnotations;
21  
22      public AnnotationGroup(AnnotatedFieldsImpl annotatedFields, String fieldName, String groupName, List<String> annotations,
23              boolean addRemainingAnnotations) {
24          this.annotatedFields = annotatedFields;
25          this.fieldName = fieldName;
26          this.groupName = groupName;
27          this.annotations = new ArrayList<>(annotations);
28          this.addRemainingAnnotations = addRemainingAnnotations; 
29      }
30  
31      public String fieldName() {
32          return fieldName;
33      }
34  
35      public String groupName() {
36          return groupName;
37      }
38  
39      public List<Annotation> annotations() {
40          AnnotatedFieldImpl f = annotatedFields.get(fieldName);
41          return annotations.stream().map(name -> f.annotation(name)).filter(a -> a != null).collect(Collectors.toList());
42      }
43  
44      public boolean addRemainingAnnotations() {
45          return addRemainingAnnotations;
46      }
47      
48      @Override
49      public Iterator<Annotation> iterator() {
50          AnnotatedFieldImpl f = annotatedFields.get(fieldName);
51          return annotations.stream().map(name -> f.annotation(name)).iterator();
52      }
53  
54      public Stream<Annotation> stream() {
55          AnnotatedFieldImpl f = annotatedFields.get(fieldName);
56          return annotations.stream().map(name -> f.annotation(name));
57      }
58  
59  }
